// Programa de Prueba
Proc linea1()
    [
        //Define variable local
        Def(varLocal1, 1);
        PosY varLocal1;
    ];
end;

Proc posiciona(valorX, valorY)
    [
        Def(PosX, valorX);
        Def(PosY, valorY);
    ];
end;

Proc impCruz(x, y)
    [
        Down;
        Pos(x,y);
        For var1(1 to 11) Loop
            [PoxY Add(var2, var1);
            ContinueRight 9;]
        End Loop;
        Up;
        PosX Substr(x, 6);
        PosY Substr(y, 5);
        Down;
        For var1(1 to 5) Loop
            [PoxY Add(var2, var1);
            ContinueRight 9;]
        End Loop;
        Up;
        Beginning;
        ];
end;

Proc main()
    [
        // Define variable global
        Def(varGlobal1, 1);
        //Llama al procedimiento linea1
        linea1();
        //Llama al procedimiento posiciona
        posiciona(1,1);
        //El color es 1
        UseColor varGlobal1;
        //Llama al procedimiento para dibujar una Cruz
        impCruz(5,5);
    ];
End;
